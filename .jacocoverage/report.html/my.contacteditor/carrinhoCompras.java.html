    public float precoProduto(){
        custo = Float.parseFloat(Custo());
        tributos = Float.parseFloat(Tributos()); //Captura os valores de custo e tributo para imprimir para o usuário.
        
        precoProduto = custo + (custo*(tributos/100)); // Faz o calculo do proço final.
        
        return precoProduto;
    }

    private void jButton3ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton3ActionPerformed
        ArrayList<ProdutoDAO> p = new ArrayList<ProdutoDAO>();
            conexaoBD cb = new conexaoBD();
            
            try{
                
                if(comboOpcao.getSelectedItem().equals("Descrição")){
                    p = (ArrayList<ProdutoDAO>) cb.consultaTodosProdutosPorNome(pesquisaGeral.getText());
                    System.out.println("ASD: "+pesquisaGeral.getText());
                }else{
                    if(comboOpcao.getSelectedItem().equals("Código")){
                        p = (ArrayList<ProdutoDAO>) cb.consultaTodosProdutosPorCodigo(produtoCodigo.getText());
                        System.out.println("ASD: "+pesquisaGeral.getText());
                    }
                }
            } catch (SQLException ex) {
            
                Logger.getLogger(carrinhoCompras.class.getName()).log(Level.SEVERE, null, ex);
            }
            
            if(p.isEmpty()){
                JOptionPane.showMessageDialog(null, "Banco de Dados Vazio!");
                return;
            }
            
            DefaultTableModel modelo = (DefaultTableModel) tabelaConsulta.getModel();
            
            modelo.setNumRows(0);
            
            Object[] fila = new Object[modelo.getColumnCount()];
            for(int i=0; i < p.size(); i++){
                modelo.addRow(new Object[]{
                    p.get(i).getCodigo(),
                    p.get(i).getNome(),
                    p.get(i).getDescricaoReduzida(),
                    p.get(i).getFabricante(),
                    p.get(i).getCusto(),
                    p.get(i).getTributos()
                });
                
            }
    }//GEN-LAST:event_jButton3ActionPerformed

    private void jButton5ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton5ActionPerformed
        DefaultTableModel modelo1 = (DefaultTableModel) tabelaCarrinho.getModel();
        
        if(tabelaCarrinho.getSelectedRow() >=0){
            precoFinal -= Float.parseFloat(modelo1.getValueAt(tabelaCarrinho.getSelectedRow(), 3).toString().replace(",", "."));
            modelo1.removeRow(tabelaCarrinho.getSelectedRow());
            tabelaCarrinho.setModel(modelo1);
        }else{
            JOptionPane.showMessageDialog(null, "Favor selecionar uma linha");
        }
        
        precoFinalCompra.setText("R$ "+formatter.format(precoFinal));
    }//GEN-LAST:event_jButton5ActionPerformed

    private void jButton4ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton4ActionPerformed
        
        precoFinal = 0;
        
        DefaultTableModel modelo1 = (DefaultTableModel) tabelaCarrinho.getModel();
        DefaultTableModel modelo = (DefaultTableModel) tabelaConsulta.getModel();
        
        int[] index = tabelaConsulta.getSelectedRows();
        Object[] row = new Object[5];
        
        String aux;
   
    try{
        
        for(int i=0; i<index.length; i++){
            row[0] = modelo.getValueAt(index[i], 0);
            row[1] = modelo.getValueAt(index[i], 1);
            row[2] = quantidade.getText();
            row[3] = formatter.format(precoProduto() * Float.parseFloat(quantidade.getText()));
            
            modelo1.addRow(row);
        }
        
        for(int i=0; i < modelo1.getRowCount(); i++){ //Calcula o preço final da compra.
            aux = modelo1.getValueAt(i, 3).toString().replace(",", ".");
            System.out.println("PRECO TOTAL: "+aux);
            precoFinal += Float.parseFloat(aux);
        }
        
        quantidade.setText("");
        
        precoFinalCompra.setText("R$ "+formatter.format(precoFinal));
        
    }catch(NumberFormatException ex){
        JOptionPane.showMessageDialog(null, "Campo 'Quantidade' não pode ser vazio!","Atenção",JOptionPane.ERROR_MESSAGE,null);
    }
    }//GEN-LAST:event_jButton4ActionPerformed
