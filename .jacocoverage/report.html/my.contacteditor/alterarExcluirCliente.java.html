    private void consultarClienteActionPerformed(java.awt.event.ActionEvent evt) {                                              
        conexaoBD cb = new conexaoBD();
        ClienteDAO cliente = null;
        
        
         if(!cnpjCliente.getText().equals(cnpjCliente.getValue())){
            JOptionPane.showMessageDialog(null,"Campo não pode ser vazio!","Erro",JOptionPane.ERROR_MESSAGE,null);
        }else{

            try{
                cliente = cb.consultarCliente(cnpjCliente.getText());
            }
            catch(SQLException ex){
                JOptionPane.showMessageDialog(null,"CPNJ não cadastrado","Erro",JOptionPane.ERROR_MESSAGE,null);
            }

            if(cliente.getCnpj()==null){
               JOptionPane.showMessageDialog(null,"CPNJ não cadastrado","Erro",JOptionPane.ERROR_MESSAGE,null);
            }else{
            JOptionPane.showMessageDialog(null,"Encontrado com sucesso!");

            conexaoBD bd = new conexaoBD();
            
            cnpjCliente.setText(cliente.getCnpj());
            nomeCliente.setText(cliente.getNome());
            ruaCliente.setText(cliente.getRua());
            bairroCliente.setText(cliente.getBairro());
            numeroCliente.setText(cliente.getNum());
            cepCliente.setText(cliente.getCep());
            tel1Cliente.setText(cliente.getTelefone1());
            tel2Cliente.setText(cliente.getTelefone2());
            razaoSocCliente.setText(cliente.getRazao());
            inscEstCliente.setText(cliente.getInscEstadual());
            cidadeCliente.setText(cliente.getCCidade());
            estadoCliente.setText(cliente.getCEstado().toString());
            }
         }
        
    }

    private void excluirClienteActionPerformed(java.awt.event.ActionEvent evt) {                                            
        conexaoBD cb = new conexaoBD();
        ClienteDAO cliente = null;

         
        try{
             
        if((cnpjCliente.getText().equals(cnpjCliente.getValue()))){ //Verifica se o campo está preenchido
           
                   
             int resposta = JOptionPane.showConfirmDialog(null, "Deseja realmente excluir o cliente?","Atenção",JOptionPane.YES_NO_OPTION);
             
             if(resposta == JOptionPane.YES_OPTION){
                cb.excluirCliente(cnpjCliente.getText());
                        cnpjCliente.setText("");
                        nomeCliente.setText("");
                        ruaCliente.setText("");
                        bairroCliente.setText("");
                        numeroCliente.setText("");
                        cepCliente.setText("");
                        tel1Cliente.setText("");
                        tel2Cliente.setText("");
                        razaoSocCliente.setText("");
                        inscEstCliente.setText("");
                        cidadeCliente.setText("");
                        estadoCliente.setText("");
                JOptionPane.showMessageDialog(null, "Cliente excluído com sucesso!");
             }else{
                 if(resposta == JOptionPane.NO_OPTION){
                     //Somente fecha a caixa de pergunta.
                 }
             }
         }else{
            
         if((!cnpjCliente.getText().equals(cnpjCliente.getValue()))){
            JOptionPane.showMessageDialog(null, "Campo CNPJ vazio!","Erro ao excluir",JOptionPane.ERROR_MESSAGE,null);
            }
        }
            }
         catch(SQLException ex){
             JOptionPane.showMessageDialog(null, "Erro na exclusão do cliente!","Atenção",JOptionPane.ERROR_MESSAGE,null);
         }
    }                                           

    private void alterarClienteActionPerformed(java.awt.event.ActionEvent evt) {                                            
        String nome = nomeCliente.getText();
        String razao = razaoSocCliente.getText();
        String estadual = inscEstCliente.getText();
        String cnpj1 = cnpjCliente.getText();
        String rua = ruaCliente.getText();
        String bairro = bairroCliente.getText();
        String numero = numeroCliente.getText();
        String cep = cepCliente.getText();
        String cidade = cidadeCliente.getText();
        String estado = estadoCliente.getText();
        String tel1 = tel1Cliente.getText();
        String tel2 = tel2Cliente.getText();
        
        validaEntrada ve = new validaEntrada();
            /*Verifica se há campos obrig. vazios. Se não tiver, faz a inserção no BD.*/
            ClienteDAO cli = new ClienteDAO(cnpj1,estadual,nome,razao,rua,bairro,cep,numero,cidade,estado,tel1,tel2);
       

                try{
                    cli.atualizaCliente();      
                    JOptionPane.showMessageDialog(null, "Atualizado com sucesso!");
                }catch(entradaInvalidaException ex){
                    JOptionPane.showMessageDialog(null,""+ex.toString().substring(85,ex.toString().length()),"Erro",JOptionPane.ERROR_MESSAGE,null);
                }
        
    }
