    private void cadastrarFornActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_cadastrarFornActionPerformed
        String nome = nomeForn.getText();
        String cnpj1 = cnpj.getText();
        String rua = ruaForn.getText();
        String bairro = bairroForn.getText();
        String numero = numeroForn.getText();
        String cep = cepForn.getText();
        String tel1 = tel1Forn.getText();
        String tel2 = tel2Forn.getText();
        String insEst = inscEst.getText();
        String razSocial = razaoSocForn.getText();
        String cidade = cidadeForn.getText();
        String estado = estadoForn.getSelectedItem().toString();
    
            /*Verifica se hÃ¡ campos obr. vazios. Se nÃ£o tiver, faz a inserÃ§Ã£o no BD.*/
        FornecedorDAO fornec = new FornecedorDAO(cnpj1,insEst,nome,razSocial,rua,bairro,cep,numero,cidade,estado,tel1,tel2);


                try{
                    fornec.insereFornecedor();
                    JOptionPane.showMessageDialog(null, "Cadastrado com sucesso!");
                    //dispose();               
                }catch(entradaInvalidaException ex){
                    JOptionPane.showMessageDialog(null,""+ex.toString().substring(85,ex.toString().length()),"Erro",JOptionPane.ERROR_MESSAGE,null);
                }catch (SQLException ex) {
                    JOptionPane.showMessageDialog(null,""+ex,"Erro",JOptionPane.ERROR_MESSAGE,null);
            }
        
    }//GEN-LAST:event_cadastrarFornActionPerformed
