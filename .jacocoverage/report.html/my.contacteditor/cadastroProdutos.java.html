        public void preencheCampos(){
        conexaoBD cb = new conexaoBD();
        ProdutoDAO p = null;
        
        try{
            p = cb.consultarProduto(PCodigo.getText());
            
        }catch(SQLException ex){
            JOptionPane.showMessageDialog(null,"Produto não cadastrado","Erro",JOptionPane.ERROR_MESSAGE,null);
        }   
            
        if(p.getCodigo()==0){
           JOptionPane.showMessageDialog(null,"Produto "+PCodigo.getText()+"não cadastrado","Erro",JOptionPane.ERROR_MESSAGE,null); 
        }else{

            PCodigo.setText(Integer.toString(p.getCodigo()));
            Pnome.setText(p.getNome());
            PCodBarras.setText(p.getCodBarras());
            Pdesc.setText(p.getDescricao());
            PDescRed.setText(p.getDescricaoReduzida());
            PlocalEstoque.setText(p.getLocalEstoque());
            PUnVenda.setSelectedItem(p.getUnCompra());
            PEstoque.setText(Integer.toString(p.getEstoqueInicial()));
            PCusto.setText(Float.toString(p.getCusto()));
            PTributos.setText(Float.toString(p.getTributos()));
            PFabricante.setSelectedItem(p.getFabricante());
            POutros.setText(p.getOutros());
            Pcateg.setSelectedItem(p.getCategoria());
            PFab.setText(p.getDataFab());
            PVal.setText(p.getDataVal());
            PObs.setText(p.getObservacoes());
  
        }
    }

    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        ConsultaProdutos cp = new ConsultaProdutos();

        int codigo = Integer.parseInt(PCodigo.getText());
        String nome = Pnome.getText();
        String codBarras = PCodBarras.getText();
        String descricao = Pdesc.getText();
        String descReduzida = PDescRed.getText();
        String localEstoque = PlocalEstoque.getText();
        String unVenda = PUnVenda.getSelectedItem().toString();
        int estoque = Integer.parseInt(PEstoque.getText());
        float custo = Float.parseFloat(PCusto.getText().replace(",", "."));
        float tributos = Float.parseFloat(PTributos.getText().replace(",", "."));
        String fabricante = PFabricante.getSelectedItem().toString();
        String outros = POutros.getText();
        String categoria = Pcateg.getSelectedItem().toString();
        String dataFabricacao = PFab.getText();
        String dataValidade = PVal.getText();
        String observacao = PObs.getText();    
        
        ProdutoDAO prod = new ProdutoDAO(codigo, nome, descricao, descReduzida, codBarras,
                                         localEstoque, fabricante, categoria, dataFabricacao,
                                         dataValidade, unVenda, estoque, custo, tributos, observacao, outros);

        if(verificaSeEhAlteracao==0){
            try{
                prod.insereProduto();
                JOptionPane.showMessageDialog(null, "Cadastrado com sucesso!");
                        //dispose();               
                }catch(entradaInvalidaException ex){
                    JOptionPane.showMessageDialog(null,""+ex.toString().substring(85,ex.toString().length()),"Erro",JOptionPane.ERROR_MESSAGE,null);
                }catch (SQLException ex) {
                    JOptionPane.showMessageDialog(null,""+ex,"Erro",JOptionPane.ERROR_MESSAGE,null);
                }
        }else{
            if(verificaSeEhAlteracao==1){
                
             try{
                prod.atualizaProduto();
                JOptionPane.showMessageDialog(null, "Cadastrado com sucesso!");
                        //dispose();               
                }catch(entradaInvalidaException ex){
                    JOptionPane.showMessageDialog(null,""+ex.toString().substring(85,ex.toString().length()),"Erro",JOptionPane.ERROR_MESSAGE,null);
                }
            }
        }
        
    }//GEN-LAST:event_jButton1ActionPerformed
