    private void consultarFornActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_consultarFornActionPerformed
        conexaoBD cb = new conexaoBD();
        FornecedorDAO forn = null;
        
        
         if(!cnpj.getText().equals(cnpj.getValue())){
            JOptionPane.showMessageDialog(null,"Campo nÃ£o pode ser vazio!","Erro",JOptionPane.ERROR_MESSAGE,null);
        }else{

            try{
                forn = cb.consultarFornecedor(cnpj.getText());
            }
            catch(SQLException ex){
                JOptionPane.showMessageDialog(null,"CPNJ nÃ£o cadastrado","Erro",JOptionPane.ERROR_MESSAGE,null);
            }

            if(forn.getCnpj()==null){
               JOptionPane.showMessageDialog(null,"CPNJ nÃ£o cadastrado","Erro",JOptionPane.ERROR_MESSAGE,null);
            }else{
            JOptionPane.showMessageDialog(null,"Encontrado com sucesso!");

            conexaoBD bd = new conexaoBD();
            
            cnpj.setText(forn.getCnpj());
            nomeForn.setText(forn.getNome());
            ruaForn.setText(forn.getRua());
            bairroForn.setText(forn.getBairro());
            numeroForn.setText(forn.getNum());
            cepForn.setText(forn.getCep());
            tel1Forn.setText(forn.getTelefone1());
            tel2Forn.setText(forn.getTelefone2());
            razaoSocForn.setText(forn.getRazao());
            inscEst.setText(forn.getInscEstadual());
            cidadeForn.setText(forn.getFCidade());
            estadoForn.setText(forn.getFEstado().toString());
            }
         }
        
    }//GEN-LAST:event_consultarFornActionPerformed

    private void alterarFornActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_alterarFornActionPerformed
        String nome = nomeForn.getText();
        String razao = razaoSocForn.getText();
        String estadual = inscEst.getText();
        String cnpj1 = cnpj.getText();
        String rua = ruaForn.getText();
        String bairro = bairroForn.getText();
        String numero = numeroForn.getText();
        String cep = cepForn.getText();
        String cidade = cidadeForn.getText();
        String estado = estadoForn.getText();
        String tel1 = tel1Forn.getText();
        String tel2 = tel2Forn.getText();
        
        validaEntrada ve = new validaEntrada();
            /*Verifica se hÃ¡ campos obrig. vazios. Se nÃ£o tiver, faz a inserÃ§Ã£o no BD.*/
            FornecedorDAO fornec = new FornecedorDAO(cnpj1,estadual,nome,razao,rua,bairro,cep,numero,cidade,estado,tel1,tel2);
       

                try{
                    fornec.atualizaFornecedor();      
                    JOptionPane.showMessageDialog(null, "Atualizado com sucesso!");
                }catch(entradaInvalidaException ex){
                    JOptionPane.showMessageDialog(null,""+ex.toString().substring(85,ex.toString().length()),"Erro",JOptionPane.ERROR_MESSAGE,null);
